190void DungeonChamberController::HandleCombatAction(int actionIndex) {
191 auto combatChamber = static_cast<ChamberCombat*>(chamber);
192
193 switch (actionIndex) {
194 case 0: // Fight
195 {
196 // Get enemies from the chamber
197 std::vector<std::shared_ptr<Unit>> enemies;
198 for (const auto& mob : combatChamber->getMobs()) {
199 auto newMob = std::make_shared<NormalMob>();
200 *newMob = mob;
201 enemies.push_back(newMob);
202 }
203
204 // Publish event to start battle
205 EventData data;
206 data.AddParam("enemies", std::move(enemies));
207 data.AddParam("goldReward", combatChamber->getGoldReward());
208 data.AddParam("expReward", combatChamber->getExpReward());
209 EventBus::Publish(EventType::BATTLE_REQUESTED, data);
210 }
211 break;
212
213 case 1: // Run Away
214 {
215 // 50% chance to successfully escape
216 bool escaped = RNG::chance(50);
217
218 if (escaped) {
219 MessageQueue::Instance().AddMessage("You managed to escape!");
220 EventData data;
221 data.AddParam("success", true);
222 EventBus::Publish(EventType::CHAMBER_COMPLETED, data);
223 } else {
224 MessageQueue::Instance().AddMessage("Failed to escape! You must fight!");
225 HandleCombatAction(0); // Fight
226 }
227 }
228 break;
229
230 case 2: // Exit Dungeon
231 MessageQueue::Instance().AddMessage("You decided to leave the dungeon.");
232 EventBus::Publish(EventType::DUNGEON_ESCAPE_REQUESTED, EventData());
233 break;
234 }

6void DungeonPathService::GenerateDoorChoices(DungeonPathModel& model) {
7 // Clear existing door options
8 model.ClearDoorOptions();
9
10 // Determine how many doors to generate (2-4)
11 std::uniform_int_distribution<> doorCountDist(2, 4);
12 int doorCount = doorCountDist(rng);
13
14 // Check if we should force a boss chamber
15 bool forceBoss = ShouldForceBossChamber(model);
16
17 if (forceBoss) {
18 // Just one door leading to a boss
19 auto bossChamber = CreateBossChamber(model);
20 model.AddDoorOption("Boss Chamber", "A powerful enemy awaits...", std::move(bossChamber));
21 model.SetFinalChamberStatus(true);
22
23 // Publish event for boss door generation
24 EventData data;
25 data.AddParam("doorCount", 1);
26 data.AddParam("forceBoss", true);
27 EventBus::Publish(EventType::DUNGEON_DOORS_GENERATED, data);
28 return;
29 }
30
31 // Generate door options
32 for (int i = 0; i < doorCount; i++) {
33 std::string doorName;
34 std::string doorDescription;
35 std::unique_ptr<Chamber> chamber;
36
37 // Each door leads to a different type of chamber with some randomness
38 std::uniform_int_distribution<> chamberTypeDist(1, 100);
39 int chamberTypeRoll = chamberTypeDist(rng);
40
41 if (model.GetCompletedCombatChambers() < model.GetRequiredCombatChambers() &&
42 (i == 0 || chamberTypeRoll <= 40)) {
43 // Ensure at least one combat chamber option if requirements not met
44 chamber = CreateCombatChamber(model);
45 doorName = "Combat Chamber";
46 doorDescription = "Enemies await behind this door...";
47 } else if (chamberTypeRoll <= 45) {
48 chamber = CreateHealChamber(model);
49 doorName = "Healing Chamber";
50 doorDescription = "You sense restorative energy...";
51 } else if (chamberTypeRoll <= 70) {
52 chamber = CreateChestChamber(model);
53 doorName = "Treasure Chamber";

55 } else if (chamberTypeRoll <= 90) {

58 doorDescription = "You hear the clink of coins...";
59 } else {
60 // Small chance of a random chamber
61 chamber = CreateRandomChamber(model);
62 doorName = "Mysterious Chamber";
63 doorDescription = "You're not sure what lies beyond...";
64 }

76bool DungeonPathService::ShouldForceBossChamber(const DungeonPathModel& model) const {
77 // Check if we've met the requirements for a boss chamber
78 if (model.GetCompletedCombatChambers() >= model.GetRequiredCombatChambers()) {
79 // Higher chance of boss appearing as more chambers are completed
80 int extraChambers = model.GetTotalChambersVisited() - model.GetRequiredCombatChambers();
81 int bossChance = 20 + (extraChambers * 10); // 20% base + 10% per extra chamber
82 bossChance = std::min(bossChance, 90); // Cap at 90%
83
84 std::uniform_int_distribution<> dist(1, 100);
85 return dist(rng) <= bossChance;
86 }
87
88 return false;
89}

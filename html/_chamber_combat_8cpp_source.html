75void ChamberCombat::generateMob() {
76 mobs.clear();
77 if (Database::normalMobTemplate.empty()) {
78 return;
79 }
80 int minLevel = getLowerMobLevel();
81 int maxLevel = getUpperMobLevel();
82
83 // Random engine
84 std::random_device rd;
85 std::mt19937 gen(rd());
86 std::uniform_int_distribution mobQuantityPicker(1, 4);
87 std::uniform_int_distribution<> levelPicker(minLevel, maxLevel);
88 std::uniform_int_distribution<> mobPicker(
89 0, Database::normalMobTemplate.size() - 1);
90
91 double statMultiplier = getMobDifficultyScale();
92
93 // Random pick mob quantity
94 int numMobs = mobQuantityPicker(gen);
95 for (int i = 0; i < numMobs; ++i) {
96 // Random pick a mob type
97 NormalMob selectedMob = Database::normalMobTemplate[mobPicker(gen)];
98
99 // Random pick level between range
100 int randomLevel = levelPicker(gen);
101
102 selectedMob.SetLevel(randomLevel);
103 selectedMob.SetAgility(selectedMob.GetAgility() * statMultiplier);
104 selectedMob.SetStrength(selectedMob.GetStrength() * statMultiplier);
105 selectedMob.SetIntelligence(selectedMob.GetIntelligence() *
106 statMultiplier);
107
108 mobs.push_back(selectedMob);
109 }

130void ChamberCombat::chamberEnded(bool win, int baseGold, int baseExp) {
131 if (baseGold <= 0) baseGold = DEFAULT_BASE_GOLD;
132 if (baseExp <= 0) baseExp = DEFAULT_BASE_EXP;
133
134 // Get multiplier based on Rank
135 int multiplier = 1;
136 auto it = rewardMultipliers.find(getRank());
137 if (it != rewardMultipliers.end()) {
138 multiplier = it->second;
139 }
140
141 // Apply base rewards
142 goldReward = baseGold * multiplier;
143 expReward = baseExp * multiplier;
144
145 // Random engine
146 static std::random_device rd;
147 static std::mt19937 gen(rd());
148
149 // Randomize rewards (±10%)
150 int goldVarianceRange = goldReward / 10;
151 int expVarianceRange = expReward / 10;
152
153 std::uniform_int_distribution<> goldVarDist(0, goldVarianceRange);
154 std::uniform_int_distribution<> expVarDist(0, expVarianceRange);
155 std::uniform_int_distribution<> signDist(0, 1);
156
157 int goldVariance = goldVarDist(gen);
158 int expVariance = expVarDist(gen);
159
160 goldReward += (signDist(gen) == 0) ? goldVariance : -goldVariance;
161 expReward += (signDist(gen) == 0) ? expVariance : -expVariance;
162
163 // Add mob-based EXP
164 int mobTotalExp = 0;
165 for (const auto &mob : mobs) {
166 mobTotalExp += mob.GetExperienceReward();
167 }
168 expReward += mobTotalExp;
169
170 // Double rewards if double dungeon
171 if (getIsDoubleDungeon()) {
172 goldReward *= 2;
173 expReward *= 2;
174 }
175
176 // If lose, apply penalty
177 if (!win) {
178 std::uniform_int_distribution<> penaltyPercentDist(10,
179 30); // 10% - 30%
180 int goldPenaltyPercent = penaltyPercentDist(gen);
181 int expPenaltyPercent = penaltyPercentDist(gen);
182
183 goldPenalty = goldReward * goldPenaltyPercent / 100;
184 expPenalty = expReward * expPenaltyPercent / 100;
185 } else {
186 generateItemBonus();
187 goldPenalty = 0;
188 expPenalty = 0;

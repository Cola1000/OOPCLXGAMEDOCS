93void DungeonOrchestrator::CompleteCurrentChamber(bool success) {
94 if (!success) {
95 // Handle failure based on dungeon type
96 if (dungeonRank == Rank::SPECIAL) {
97 // Special dungeons allow continuing even on failure
98 currentChamber.reset();
99 chamberController.reset();

102 } else {
103 // Normal dungeons end on failure
104 CompleteDungeon(false);

108
109 // Chamber completed successfully
110 currentChamber.reset();
111 chamberController.reset();

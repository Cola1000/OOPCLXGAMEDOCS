110void ChamberBoss::chamberEnded(bool win, int baseGold, int baseExp) {
111 if (baseGold <= 0) baseGold = DEFAULT_BASE_GOLD_BOSS;
112 if (baseExp <= 0) baseExp = DEFAULT_BASE_EXP_BOSS;
113
114 // Get multiplier based on Rank
115 int multiplier = 1;
116 auto it = rewardMultiplierBoss.find(getRank());
117 if (it != rewardMultiplierBoss.end()) {
118 multiplier = it->second;
119 }
120
121 // Apply base rewards
122 goldReward = baseGold * multiplier;
123 expReward = baseExp * multiplier;
124
125 // Random engine
126 static std::random_device rd;
127 static std::mt19937 gen(rd());
128
129 // Randomize rewards (±10%)
130 int goldVarianceRange = goldReward / 10;
131 int expVarianceRange = expReward / 10;
132
133 std::uniform_int_distribution<> goldVarDist(0, goldVarianceRange);
134 std::uniform_int_distribution<> expVarDist(0, expVarianceRange);
135 std::uniform_int_distribution<> signDist(0, 1);
136
137 int goldVariance = goldVarDist(gen);
138 int expVariance = expVarDist(gen);
139
140 goldReward += (signDist(gen) == 0) ? goldVariance : -goldVariance;
141 expReward += (signDist(gen) == 0) ? expVariance : -expVariance;
142
143 // Boss exp reward
144 expReward += bossMob.GetExperienceReward();
145
146 // Double rewards if dungeon is a double dungeon
147 if (getIsDoubleDungeon()) {
148 goldReward *= 2;
149 expReward *= 2;
150 }
151
152 // If lose, apply penalty
153 if (!win) {
154 std::uniform_int_distribution<> penaltyPercentDist(10,
155 30); // 10% - 30%
156 int goldPenaltyPercent = penaltyPercentDist(gen);
157 int expPenaltyPercent = penaltyPercentDist(gen);
158
159 goldPenalty = goldReward * goldPenaltyPercent / 100;
160 expPenalty = expReward * expPenaltyPercent / 100;
161 } else {
162 generateItemBonus();
163 goldPenalty = 0;
164 expPenalty = 0;
165 }

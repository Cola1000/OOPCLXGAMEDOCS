167bool Dungeon::addChamber() {
168 if (chambers.size() >= static_cast<size_t>(totalChamber)) {

171
172 static std::random_device rd;
173 static std::mt19937 gen(rd());
174
175 if (!isDoubleDungeon) {
176 initRandomDoubleDungeon();
177 }
178
179 int position = chambers.size() + 1;
180 bool isFinalChamber = (position == totalChamber);
181
182 std::uniform_int_distribution<> chamberTypeDist(1, 100);
183 int chamberTypeRoll = chamberTypeDist(gen);
184
185 if (isFinalChamber) {
186 ChamberBoss newChamber(rank, lowerMobLevel, upperMobLevel,
187 mobDifficultyScale, isDoubleDungeon);
188 chambers.push_back(newChamber);
189 } else if (chamberTypeRoll <= 50 || position == totalChamber - 1) {
190 ChamberCombat newChamber(rank, lowerMobLevel, upperMobLevel,
191 mobDifficultyScale, isDoubleDungeon);
192 isChamberCombatCreated = true;
193 chambers.push_back(newChamber);
194 } else if (chamberTypeRoll > 50 && chamberTypeRoll <= 70) {
195 ChamberHeal newChamber(rank, isDoubleDungeon);
196 } else if (chamberTypeRoll > 70 && chamberTypeRoll <= 90) {
197 ChamberShop newChamber(rank, isDoubleDungeon);
198 } else if (chamberTypeRoll > 90 && chamberTypeRoll <= 100) {
199 ChamberChest newChamber(rank, isDoubleDungeon);
200 chambers.push_back(newChamber);

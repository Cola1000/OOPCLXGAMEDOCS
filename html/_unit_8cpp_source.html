79void Unit::Attack(Unit& target) {
80 int hitPoints = CalculateHitPoints();
81 int evadePoints = target.CalculateEvadePoints();
82 int hitMargin = hitPoints - evadePoints;
83
84 bool isCrit = (hitMargin > 10);
85 bool isHit = (hitMargin >= 0);
86
87 if (isHit) {
88 int damage = CalculateAttackDamage();
89 if (damage > 0) {
90 if (isCrit) {
91 damage = static_cast<int>(damage * 1.5); // Default critical multiplier
92 }
93 int damageDealt = target.TakeDamage(damage);
94 ApplyEffectOnAttack(damageDealt, &target);
95 }
96 }
97}
